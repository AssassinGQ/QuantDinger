```python
import pytest
from unittest.mock import Mock, patch, MagicMock

class TestIBKRIntegration:
    """IBKR 集成测试"""
    
    def test_factory_exchange_id_param(self):
        """测试 DataSourceFactory 接受 exchange_id 参数 (INT-01)"""
        from app.data_sources.factory import DataSourceFactory
        import inspect
        sig = inspect.signature(DataSourceFactory.get_source)
        assert 'exchange_id' in sig.parameters
    
    def test_ibkr_live_returns_ibkr_datasource(self):
        """测试 exchange_id='ibkr-live' 返回 IBKRDataSource (INT-03)"""
        from app.data_sources.factory import DataSourceFactory
        # 清理现有实例
        DataSourceFactory._sources.clear()
        
        with patch('app.data_sources.factory.IBKRDataSource') as mock_ibkr:
            mock_instance = MagicMock()
            mock_ibkr.return_value = mock_instance
            
            ds = DataSourceFactory.get_source("Crypto", exchange_id="ibkr-live")
            
            mock_ibkr.assert_called_once()
            assert ds == mock_instance
    
    def test_exchange_id_priority_over_market(self):
        """测试 exchange_id 优先于 market (D-08)"""
        from app.data_sources.factory import DataSourceFactory
        DataSourceFactory._sources.clear()
        
        with patch('app.data_sources.factory.IBKRDataSource') as mock_ibkr:
            mock_instance = MagicMock()
            mock_ibkr.return_value = mock_instance
            
            # 即使 market 是 Crypto，exchange_id='ibkr-live' 也应返回 IBKRDataSource
            ds = DataSourceFactory.get_source("Crypto", exchange_id="ibkr-live")
            
            mock_ibkr.assert_called_once()
    
    def test_backward_compatible_no_exchange_id(self):
        """测试向后兼容: 不传 exchange_id 时按 market 处理 (D-03)"""
        from app.data_sources.factory import DataSourceFactory
        DataSourceFactory._sources.clear()
        
        # 不传 exchange_id，应该返回对应 market 的数据源
        # 这里测试 Crypto 返回其他数据源，不是 IBKRDataSource
        with patch('app.data_sources.factory.DataSourceFactory._create_source') as mock_create:
            mock_source = MagicMock()
            mock_create.return_value = mock_source
            
            ds = DataSourceFactory.get_source("Crypto")
            
            mock_create.assert_called_once_with("Crypto")
```
